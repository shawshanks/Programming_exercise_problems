## 寻找目标在序列中的出现次数 p23
```python
def count(data, target):
    n = 0 
    for i in len(data):
        if data[i] == target:
          n += 1
    return n
```

## 寻找序列中的最大值 p21 p22
```python
##方法一 检查列表中的值
def find_max1(data):
    biggest = data[0]
    for val in data:
        if val > biggest:
            biggest = val
    return biggest
       
       
##方法二 检查列表中的索引
def find_max2(data):
    big_index = 0
    for i in range(len(data)):
        if data[i] > data[big_index]:
            big_index = i
    return data[big_index]
            
```

## 寻找某个数的因数 p40

#方法1: for 循环
def factors(n):
    result = []
    for k in range(1, n):
        if n % k == 0:
            result.append(k)
    return result
 
# 方法2： 列表推导式
result = [k for k in range(1, n+1) if n % k == 0] # 在P43 1.9.2节中出现
    
 # 方法3： while 循环
def factor_while(n):
    result = []
    k = 1
    while k * k < n:
        if n % k == 0:
            result.append(k)
            result.append(n // k)
            k += 1
    if k * k == n:
        result.append(k)

 def fibonacci():
    a , b = 0, 1
    while True:
        yield a 
        future = a + b
        a = b 
        b = future         # yield语句后面可以替换为： a, b = b, a+b
